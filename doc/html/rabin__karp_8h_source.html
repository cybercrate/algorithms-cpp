23// Converts a string to an integer.
24auto create_hash = [](const std::string& text, int count)
25{
26 const auto prime{5};
27 std::int64_t result{};

35// Re-hash a string using known existing hash.
36auto recalculate_hash = [](
37 const std::string& text,

39 std::size_t new_index,
40 std::int64_t hash,
41 std::size_t pattern_length)
42{
43 const auto prime{5};

53// Compare if two sub-strings are equal.
54auto check_for_equal = [](
55 const std::string& first_text,
56 const std::string& second_text,
57 std::size_t first_start,
58 std::size_t first_end,
59 std::size_t second_start,
60 std::size_t second_end)
61{
62 if ((first_end - first_start) != (second_end - second_start))
63 return false;
64
65 while ((first_end > first_start) && (second_end > second_start)) {
66 if (first_text[first_start] != second_text[second_start])
67 return false;
68
69 ++first_start;
70 ++second_start;
71 }
72 return true;

85std::optional<std::size_t> rabin_karp(const std::string& text, const std::string& pattern)
86{
87 auto pat_hash = create_hash(pattern, pattern.size());
88 auto str_hash = create_hash(text, pattern.size());
89
90 for (std::size_t i = 0; i <= (text.size() - pattern.size()); ++i) {
91 auto equal = check_for_equal(text, pattern, i, i + pattern.size() - 1, 0, pattern.size() - 1);
92
93 if ((pat_hash == str_hash) && equal)
94 return i;
95
96 if (i < text.size() - pattern.size())
97 str_hash = recalculate_hash(text, i, i + pattern.size(), str_hash, pattern.size());
98 }
99 return std::nullopt;
